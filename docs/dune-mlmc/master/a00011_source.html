91  void assignProcessors(MPI_Comm world = MPI_COMM_WORLD) {
92  check(_diff != NULL, "No Difference object set.");
93  int size, rank;
94  int range[1][3];
95  MPI_Group all, group, masters;
96  MPI_Comm_size(world, &size);
97  MPI_Comm_rank(world, &rank);
98  MPI_Comm_group(world, &all);
99  range[0][0] = rank - rank % _p;
100  range[0][1] = range[0][0] + _p - 1;
101  range[0][2] = 1;
102  MPI_Group_range_incl(all, 1, range, &group);
103  MPI_Comm_create(world, group, &_group);
104  _g = size / _p;
105  _diff->init(world, _group);
106  range[0][0] = 0;
107  range[0][1] = size - 1;
108  range[0][2] = _p;
109  MPI_Group_range_incl(all, 1, range, &masters);
110  MPI_Comm_create(world, masters, &_masters);
111  _isMaster = (rank % _p) == 0;
112  }
